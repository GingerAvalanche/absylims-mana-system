quarterly_playable_pulse = {
	events = {
		AMS_mana_events.001
		delay = { months = 1 }
		AMS_mana_events.001
		delay = { months = 2 }
		AMS_mana_events.001
	}
}

# Fire this action to make updates to the player's mana and mana gen
on_reset_mana_system = {
	trigger = { has_trait = mana }

	effect = {
		if = {
			limit = { NOT = { exists = var:mana } }
			set_variable = {
				name = mana
				value = 0
			}
		}
		set_variable = {
			name = max_mana
			value = 0
		}
		set_variable = {
			name = mana_gen
			value = 0
		}
		trigger_event = { on_action = reset_mana_system }
		if = {
			limit = { var:mana.compare_value > var:max_mana.compare_value }
			set_variable = {
				name = mana
				value = var:max_mana
			}
		}
	}
}
